2.13 Application Logic Flowchart

flowchart TD
    START(["HTTP Request"]) --> CORS{"CORS Check"}
    CORS -->|Blocked| ERR1["403 Error"]
    CORS -->|OK| RL{"Rate Limit?"}
    RL -->|Exceeded| ERR2["429 Error"]
    RL -->|OK| ROUTE{"Route Match?"}
    ROUTE -->|No| ERR3["404 Error"]
    ROUTE -->|Login| LOGIN["Login Handler"]
    ROUTE -->|Protected| AUTH["JWT Middleware"]
    LOGIN --> FIND["Find User"]
    FIND --> MATCH{"Password OK?"}
    MATCH -->|No| ERR4["401 Error"]
    MATCH -->|Yes| TOKEN["Sign JWT"]
    TOKEN --> RES1["200 Response"]
    AUTH --> VERIFY{"Token Valid?"}
    VERIFY -->|No| ERR5["401 Error"]
    VERIFY -->|Yes| RBAC{"Role Check"}
    RBAC -->|Denied| ERR6["403 Error"]
    RBAC -->|OK| HANDLER["Route Handler"]
    HANDLER --> BIZ["Business Logic"]
    BIZ --> DBOP["DB Operation"]
    DBOP --> AUDIT["Log Audit"]
    AUDIT --> RESP["200 Response"]
